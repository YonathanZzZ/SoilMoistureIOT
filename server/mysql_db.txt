create table Users (email VARCHAR(255) PRIMARY KEY NOT NULL, password CHAR(60) NOT NULL);

create table Devices (id CHAR(36) PRIMARY KEY NOT NULL, email VARCHAR(255) NOT NULL, FOREIGN KEY (email) REFERENCES Users(email));

create table Measurements (id INT PRIMARY KEY AUTO_INCREMENT NOT NULL, deviceID CHAR(36) NOT NULL, moisture_percentage TINYINT NOT NULL, FOREIGN KEY (deviceID) REFERENCES Devices(id));
